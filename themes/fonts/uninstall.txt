@echo off

rem 获取当前路径
set cur_dir=%CD%

rem 列出当前路径下的所有字体文件
set font_files=!cur_dir!\*.ttf !cur_dir!\*.otf

rem 循环删除字体文件
for %%f in (%font_files%) do (
  rem 使用 PowerShell 删除字体文件
  pwsh -c "Remove-Item '%%f'"
)

rem 判断脚本是否成功
if %errorlevel% == 0 (
  echo "脚本运行成功！"
) else (
  echo "脚本运行失败！"
)
